#include <stdio.h>

float diametro(float raio) {
    return raio*2;
}
float circunferencia(float raio) {
    return 3.14159*2*raio;
}
float area(float raio) {
    return 3.14159*raio*raio;
}
int main() {
    float raio;
    printf("Digite o raio: ");
    scanf("%f", &raio);
    
    printf("\nO raio e: %.2f", raio);
    printf("\nO diametro e: %.2f", diametro(raio));
    printf("\nA circunferencia e: %.2f", circunferencia(raio));
    printf("\nA area e: %.2f", area(raio));

    return 0;
}
