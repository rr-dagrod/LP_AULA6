#include <stdio.h>
#include <math.h>

float juros_compostos (float valInvestido, float taxaMes, int tempoLimite) {
    float montante;
    montante = valInvestido * pow((1 + taxaMes), tempoLimite);
    return montante;
}

void main () {
    float _valInvestido, _taxaMes, _montante;
    int _tempoLimite;
    
    printf("Insira o capital inicial: ");
    scanf("%f", &_valInvestido);

    printf("Insira a taxa: ");
    scanf("%f", &_taxaMes);
    _taxaMes = _taxaMes / 100;
    
    printf("Insira o tempo no qual o dinheiro ficará investido: ");
    scanf("%d", &_tempoLimite);
    
    _montante = juros_compostos(_valInvestido, _taxaMes, _tempoLimite);

    printf("O total de Juros compostos recebido é: %.2f", (_montante-_valInvestido));
}
